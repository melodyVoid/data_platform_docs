# 项目结构

整个数据平台项目的目录结构如下：

├── docs                            // 文档
├── public                          // HTML 和静态文件
├── src                             // 源代码
│   ├── api                         // 所有请求
│   ├── assets                      // 静态资源
│   ├── components                  // 全局公用组件
│   ├── constant                    // 配置数据/常量
│   ├── icons                       // 项目所有 svg icons
│   ├── layout                      // 布局
│   ├── libs                        // 第三方资源
│   ├── request                     // 请求方法
│   ├── router                      // 路由
│   ├── store                       // 全局 store管理
│   ├── styles                      // 全局样式
│   ├── utils                       // 全局公用方法
│   ├── vendor                      // 公用vendor
│   ├── views                       // views 所有页面
│   ├── App.vue                     // 入口页面
│   ├── main.js                     // 入口文件 加载组件 初始化等
│   └── permission.js               // 权限管理
├── .browserslistrc                 // 浏览器列表
├── .env.development                // 开发环境配置
├── .env.production                 // 生产环境配置
├── .env.staging                    // 开发环境下构建配置
├── .eslintrc.js                    // eslint 配置项
├── .gitignore                      // git 忽略项
├── .prettierrc.js                  // prettierrc 配置
├── .babel.config.js                // babel-loader 配置
├── deploy.sh                       // 文档部署脚本
├── .lint-staged.config.js          // lint-staged 配置
├── package.json                    // 工程文件
├── postcss.config.js               // postcss 配置项
├── README.md                       // README
├── tsconfig.json                   // ts 配置
└── vue.config.js                   // vue 配置

其中 views、router、constant和 api 里面的结构基本一致，可一一对应，下面只展示一部分。

├── views                           // 所有页面
│   ├── ...
│   ├── userData                    // 用户数据模块
│   │   ├── profile                 // 用户画像
│   │   ├── type                    // 新增激活
│   │   └── statistics              // 付费统计
│   ├── premiere                    // 首播模块
│   │   ├── aggregateStatistics     // 整体统计
│   │   ├── courseDetail            // 课程详情
│   │   ├── nps                     // NPS 评分
│   │   └── attendanceRate          // 到/完课率
│   ├── playback                    // 回放模块
│   │   ├── aggregateStatistics     // 整体统计
│   │   ├── course                  // 课程列表
│   │   ├── courseDetail            // 课程详情
└── └── └── distribution            // 分布情况

├── api                             // api 请求
│   ├── ...
│   ├── userData                    // 用户数据模块请求
│   │   ├── profile.js              // 用户画像请求
│   │   ├── type.js                 // 新增激活请求
│   │   └── statistics.js           // 付费统计请求
│   ├── premiere                    // 首播模块请求
│   │   ├── aggregateStatistics.js  // 整体统计请求
│   │   ├── courseDetail.js         // 课程详情请求
│   │   ├── nps.js                  // NPS 评分请求
│   │   └── attendanceRate.js       // 到/完课率请求
│   ├── playback                    // 回放模块请求
│   │   ├── aggregateStatistics.js  // 整体统计请求
│   │   ├── course .js              // 回放课程列表请求
│   │   ├── courseDetail.js         // 回放课程详情请求
└── └── └── distribution.js         // 分布情况请求

├── constant                        // 配置数据/常量
│   ├── ...
│   ├── userData                    // 用户数据里的数据配置/常量
│   │   ├── profile.js              // 用户画像数据配置/常量
│   │   ├── type.js                 // 新增激活数据配置/常量
│   │   └── statistics.js           // 付费统计数据配置/常量
│   ├── premiere                    // 首播模块数据配置/常量
│   │   ├── aggregateStatistics.js  // 整体统计数据配置/常量
│   │   ├── courseDetail.js         // 课程详情数据配置/常量
│   │   ├── nps.js                  // NPS 评分数据配置/常量
│   │   └── attendanceRate.js       // 到/完课率数据配置/常量
│   ├── playback                    // 回放模块数据配置/常量
│   │   ├── aggregateStatistics.js  // 整体统计数据配置/常量
│   │   ├── course .js              // 回放课程列表数据配置/常量
│   │   ├── courseDetail.js         // 回放课程详情数据配置/常量
└── └── └── distribution.js         // 分布情况数据配置/常量

├── router                          // 路由
│   ├── ...
│   ├── index.js                    // 路由入口文件
│   ├── userData                    // 用户数据模块路由
│   ├── premiere                    // 首播模块路由
└── └── playback                    // 回放模块路由
