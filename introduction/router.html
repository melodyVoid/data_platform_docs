# 路由

路由文件夹项目结构

├── router                          // 路由
│   ├── index.js                    // 路由入口文件
│   ├── common                      // 通用路由（login/dashboard/404等）
│   │   └── index.js
│   ├── userData                    // 用户数据模块路由
│   │   └── index.js
│   ├── premiere                    // 首播模块路由
│   │   └── index.js
│   ├── playback                    // 回放模块路由
│   │   └── index.js
│   ├── transformation              // 传播转化模块路由
│   │   └── index.js
│   ├── yanzhi                      // 研值模块路由
│   │   └── index.js
│   ├── creditAndDegree             // 学分/学位模块路由
│   │   └── index.js
│   ├── renew                       // 续费率模块路由
└── └── └── index.js

路由配置项

//当设置 true 的时候该路由不会再侧边栏出现 如404，login等页面。
hidden: true // (默认 false)

//当设置 noredirect 的时候该路由在面包屑导航中不可被点击
redirect: noredirect

name: 'router-name' // 设定路由的名字。
meta: {
  title: 'title' // 设置该路由在侧边栏和面包屑中展示的名字
  icon: 'svg-name' // 设置该路由的图标
}

示例

{
    path: '/premiere',
    component: Layout,
    redirect: '/premiere/aggregate',
    name: 'Premiere',
    meta: { title: '首播', icon: 'premiere' },
    children: [
      {
        path: 'aggregate',
        name: 'PremiereAggregate',
        // 路由懒加载
        component: () => 
          import(/* webpackChunkName: 'premiereAggregate' */ 'src/views/premiere/aggregateStatistics'),
        meta: { title: '整体统计', icon: 'aggregate' }
      },
      {
        path: 'nps',
        name: 'NPS',
        component: () => 
          import(/* webpackChunkName: 'nps' */ 'src/views/premiere/nps'),
        meta: { title: 'NPS 评分', icon: 'nps' }
      },
      {
        path: 'attendance',
        name: 'attendanceRate',
        component: () => 
          import(/* webpackChunkName: 'attendance' */ 'src/views/premiere/attendanceRate'),
        meta: { title: '到/完课率', icon: 'attendanceRate' }
      },
      {
        path: 'course_detail/:course_id',
        name: 'course_detail',
        component: () => 
          import(/* webpackChunkName: 'course_detail' */ 'src/views/premiere/courseDetail'),
        hidden: true
      }
    ]
  },

注：图标为自己下载的 svg 文件，存放在 src/icons/svg 里面。
