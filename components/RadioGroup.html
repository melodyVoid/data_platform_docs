    # 单选按钮组

    # 描述 [Description]

    通用单选按钮组，可传入一个数组来决定选项的内容。在页面的头部（用于切换类别）和个别图表上部（用于切换图表指标）均有出现。

    # 用法 [Usage]

    <radio-group
  v-model="value"
  :data="radioList"
  @change="handleChange"
></radio-group>

    # 属性 [Attributes]

    # 方法 [Events]

    # 例子 [Example]

    value: regional 
    this.radioList = [
  { label: '地域', value: 'regional' },
  { label: '性别', value: 'gender' },
  { label: '行业', value: 'industry' },
  { label: '职位', value: 'position' },
  { label: '企业规模', value: 'scale' },
  { label: '工作年限', value: 'seniority' },
]
