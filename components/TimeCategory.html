    # 时间类别

    # 描述 [Description]

    时间类别组件，「用户类型」和「付费统计」中有用到，和日期范围联动，日期范围不满足"周"和"月"时分别置灰不可点击。

    # 用法 [Usage]

    <time-category
  v-model="time_category"
  :date="date_range"
  @change="changeTimeCategory"
></time-category>

    # 属性 [Attributes]

    # 方法 [Events]

    # 例子 [Example]

    -
    day
    <date-range
  v-model="date_range"
  @init="initDate"
></date-range>

<time-category
  v-model="time_category"
  :date="date_range"
  @change="changeTimeCategory"
></time-category>

...
<script>
export default {
  data() {
    return {
      date_range: [],
      time_category: 'day'
    }
  },
  methods: {
    initDate(date_range) {
      this.date_range = date_range
    },
    changeTimeCategory(time_category) {
      console.log(time_category)
      this.$notify({
        title: '触发事件',
        type: 'success',
        duration: 2000,
        message: '触发了 change 事件，返回值为：' + time_category
      })
    }
  }
}
</script>
